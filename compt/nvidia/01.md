def parse_xml_field(text: str, tag: str) -> str:
    """Extract content between <tag>...</tag>."""
    pattern = rf"<{tag}>(.*?)</{tag}>"
    match = re.search(pattern, text, re.DOTALL)
    return match.group(1).strip() if match else ""

def load_qa_data(path: str) -> list:
    """
    Expects JSONL where each line has either:
      - separate fields: {"question": "...", "thinking": "...", "answer": "..."}
      - or a single 'text' field with XML tags
    """
    data = []
    with open(path, "r") as f:
        for line in f:
            obj = json.loads(line.strip())
            if "question" in obj:
                data.append({
                    "question": obj["question"],
                    "thinking": obj.get("thinking", ""),
                    "answer":   obj["answer"],
                })
            elif "text" in obj:
                txt = obj["text"]
                data.append({
                    "question": parse_xml_field(txt, "question"),
                    "thinking": parse_xml_field(txt, "thinking"),
                    "answer":   parse_xml_field(txt, "answer"),
                })
    return data

# ── 4a.  Model + Tokenizer loader ─────────────────────────────────────────────

def load_base_model_and_tokenizer():
    """Load quantized base model + tokenizer."""
    bnb_config = None
    if cfg.load_in_4bit:
        bnb_config = BitsAndBytesConfig(
            load_in_4bit=True,
            bnb_4bit_quant_type="nf4",
            bnb_4bit_compute_dtype=getattr(torch, cfg.dtype),
            bnb_4bit_use_double_quant=True,
        )
    model = AutoModelForCausalLM.from_pretrained(
        cfg.base_model_name,
        quantization_config=bnb_config,
        device_map=cfg.device_map,
        torch_dtype=getattr(torch, cfg.dtype),
        trust_remote_code=True,
        attn_implementation="flash_attention_2",   # remove if not supported
    )
    tokenizer = AutoTokenizer.from_pretrained(
        cfg.base_model_name,
        trust_remote_code=True,
        padding_side="left",          # important for batch generation
    )
    if tokenizer.pad_token is None:
        tokenizer.pad_token = tokenizer.eos_token
        model.config.pad_token_id = tokenizer.eos_token_id
    return model, tokenizer


def attach_lora(model, checkpoint_path: str = None):
    """Attach fresh LoRA or load from checkpoint."""
    if checkpoint_path and os.path.exists(checkpoint_path):
        print(f"   Loading LoRA from {checkpoint_path}")
        model = PeftModel.from_pretrained(model, checkpoint_path)
    else:
        if cfg.load_in_4bit:
            model = prepare_model_for_kbit_training(model)
        lora_cfg = LoraConfig(
            r=cfg.lora_r,
            lora_alpha=cfg.lora_alpha,
            lora_dropout=cfg.lora_dropout,
            target_modules=cfg.lora_target_modules,
            bias="none",
            task_type=TaskType.CAUSAL_LM,
        )
        model = get_peft_model(model, lora_cfg)
    model.print_trainable_parameters()
    return model


# ── 4b.  Chat formatting ──────────────────────────────────────────────────────

def format_chat(question: str, system_prompt: str, thinking: str = "", answer: str = ""):
    """Build chat messages list (OpenAI-style)."""
    messages = [{"role": "system", "content": system_prompt}]
    messages.append({"role": "user", "content": question})
    if answer:
        assistant_content = ""
        if thinking:
            assistant_content += f"<thinking>\n{thinking}\n</thinking>\n\n"
        assistant_content += f"<answer>\n{answer}\n</answer>"
        messages.append({"role": "assistant", "content": assistant_content})
    return messages


# ── 4c.  Generation ───────────────────────────────────────────────────────────

@torch.inference_mode()
def generate_answer(model, tokenizer, question: str, system_prompt: str,
                    temperature: float = 0.0, max_new_tokens: int = 512) -> str:
    """Generate a single answer."""
    messages = format_chat(question, system_prompt)
    text = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
    inputs = tokenizer(text, return_tensors="pt").to(model.device)
    
    gen_kwargs = dict(
        max_new_tokens=max_new_tokens,
        do_sample=(temperature > 0),
        temperature=max(temperature, 1e-7),
        top_p=0.95,
        pad_token_id=tokenizer.pad_token_id,
    )
    out = model.generate(**inputs, **gen_kwargs)
    new_tokens = out[0, inputs["input_ids"].shape[1]:]
    return tokenizer.decode(new_tokens, skip_special_tokens=True)


@torch.inference_mode()
def generate_batch(model, tokenizer, questions: list, system_prompt: str,
                   temperature: float = 0.0, max_new_tokens: int = 512,
                   batch_size: int = 8) -> list:
    """Batch generation for speed."""
    all_outputs = []
    for i in range(0, len(questions), batch_size):
        batch_q = questions[i:i+batch_size]
        texts = []
        for q in batch_q:
            msgs = format_chat(q, system_prompt)
            texts.append(tokenizer.apply_chat_template(
                msgs, tokenize=False, add_generation_prompt=True))
        
        inputs = tokenizer(texts, return_tensors="pt", padding=True,
                           truncation=True, max_length=cfg.max_seq_length).to(model.device)
        gen_kwargs = dict(
            max_new_tokens=max_new_tokens,
            do_sample=(temperature > 0),
            temperature=max(temperature, 1e-7),
            top_p=0.95,
            pad_token_id=tokenizer.pad_token_id,
        )
        out = model.generate(**inputs, **gen_kwargs)
        for j, seq in enumerate(out):
            new_tokens = seq[inputs["input_ids"].shape[1]:]
            all_outputs.append(tokenizer.decode(new_tokens, skip_special_tokens=True))
    return all_outputs


# ── 4d.  Answer extraction & comparison ───────────────────────────────────────

def extract_answer(text: str) -> str:
    """Pull text from <answer>...</answer> tags, fallback to full text."""
    ans = parse_xml_field(text, "answer")
    if ans:
        return ans.strip()
    # fallback: take last line or full text
    return text.strip().split("\n")[-1].strip()

def normalize_answer(ans: str) -> str:
    """Normalize for comparison."""
    ans = ans.lower().strip()
    # remove common LaTeX wrappers
    ans = re.sub(r"\$|\\boxed\{|\}|\\text\{", "", ans)
    # remove punctuation
    ans = re.sub(r"[^\w\s\.\-]", "", ans)
    return ans.strip()

def is_correct(predicted: str, gold: str) -> bool:
    """Check if predicted answer matches gold."""
    return normalize_answer(predicted) == normalize_answer(gold)


# ── 4e.  Accuracy measurement ────────────────────────────────────────────────

def measure_accuracy(model, tokenizer, dataset: list, system_prompt: str,
                     temperature: float = 0.0, label: str = "eval",
                     max_samples: int = None) -> dict:
    """Measure accuracy on a dataset. Returns dict with metrics."""
    if max_samples:
        dataset = dataset[:max_samples]
    
    questions = [d["question"] for d in dataset]
    gold_answers = [d["answer"] for d in dataset]
    
    print(f"   Generating {len(questions)} answers for [{label}]…")
    predictions = generate_batch(model, tokenizer, questions, system_prompt,
                                  temperature=temperature)
    
    extracted = [extract_answer(p) for p in predictions]
    correct = [is_correct(e, g) for e, g in zip(extracted, gold_answers)]
    acc = sum(correct) / len(correct)
    
    # check format compliance
    format_ok = [("<thinking>" in p and "<answer>" in p) for p in predictions]
    format_rate = sum(format_ok) / len(format_ok)
    
    # average response length
    avg_len = np.mean([len(p.split()) for p in predictions])
    
    result = {
        "label": label,
        "accuracy": round(acc, 4),
        "format_compliance": round(format_rate, 4),
        "avg_response_words": round(avg_len, 1),
        "n_samples": len(dataset),
        "n_correct": sum(correct),
    }
    print(f"   ✅  [{label}]  accuracy={acc:.4f}  format={format_rate:.4f}  avg_len={avg_len:.0f}")
    return result


# ── 4f.  Reward function (for RL steps) ──────────────────────────────────────

def compute_reward(response: str, gold_answer: str) -> float:
    """
    Multi-signal reward with anti-hacking measures.
    """
    reward = 0.0
    
    # 1) Correctness — primary signal
    pred = extract_answer(response)
    if is_correct(pred, gold_answer):
        reward += cfg.reward_correct
    else:
        reward += cfg.reward_incorrect
    
    # 2) Format bonus — encourages structured output
    has_thinking = "<thinking>" in response and "</thinking>" in response
    has_answer   = "<answer>" in response and "</answer>" in response
    if has_thinking and has_answer:
        reward += cfg.reward_format_bonus
    
    # 3) Length penalty — discourages verbose reward hacking
    n_tokens = len(response.split())
    reward -= cfg.reward_length_penalty * n_tokens
    
    # 4) Repetition penalty — detect copy-paste / looping
    sentences = response.split(".")
    if len(sentences) > 3:
        unique_ratio = len(set(sentences)) / len(sentences)
        if unique_ratio < 0.5:
            reward -= cfg.reward_repetition_penalty
    
    return reward


# ── 4g.  Results tracker ────────────────────────────────────────────────────

class ResultsTracker:
    """Collect accuracy across all steps for final comparison."""
    def __init__(self):
        self.results = {}
    
    def add(self, step: str, metrics: dict):
        self.results[step] = metrics
        
    def summary(self):
        print("\n" + "="*80)
        print("  RESULTS SUMMARY")
        print("="*80)
        print(f"  {'Step':<35} {'Accuracy':>10} {'Format':>10} {'AvgLen':>10}")
        print("-"*80)
        for step, m in self.results.items():
            print(f"  {step:<35} {m['accuracy']:>10.4f} {m['format_compliance']:>10.4f} {m['avg_response_words']:>10.1f}")
        print("="*80)

tracker = ResultsTracker()

# ── 4h.  Memory cleanup ──────────────────────────────────────────────────────

def cleanup(*models):
    for m in models:
        if m is not None:
            del m
    gc.collect()
    torch.cuda.empty_cache()

print("✅  Helper functions defined")

def build_sft_dataset(data: list, system_prompt: str) -> HFDataset:
    """Convert to HF Dataset with 'text' field ready for SFTTrainer."""
    rows = []

# ── Elastic Weight Consolidation (EWC) — catastrophic forgetting guard ────────

class EWCRegularizer:
    """
    Approximate EWC: store Fisher diagonal + param snapshot from SFT checkpoint.
    Adds penalty:  λ/2 · Σ F_i (θ_i - θ*_i)^2
    """
    def __init__(self, model, dataloader, n_samples: int = 200, lam: float = 5000.0):
        self.lam = lam
        self.params = {}
        self.fisher = {}
        
        if not cfg.ewc_enabled:
            return
        
        print("   Computing Fisher Information for EWC…")
        model.eval()
        fisher_acc = {}
        for name, p in model.named_parameters():
            if p.requires_grad:
                self.params[name] = p.data.clone()
                fisher_acc[name] = torch.zeros_like(p.data)
        
        count = 0
        for batch in dataloader:
            if count >= n_samples:
                break
            inputs = {k: v.to(model.device) for k, v in batch.items()}
            outputs = model(**inputs, labels=inputs.get("input_ids"))
            loss = outputs.loss
            loss.backward()
            for name, p in model.named_parameters():
                if p.requires_grad and p.grad is not None:
                    fisher_acc[name] += p.grad.data ** 2
            model.zero_grad()
            count += inputs["input_ids"].shape[0]
        
        for name in fisher_acc:
            self.fisher[name] = fisher_acc[name] / max(count, 1)
        model.train()
    
    def penalty(self, model) -> torch.Tensor:
        if not cfg.ewc_enabled or not self.fisher:
            return torch.tensor(0.0, device=model.device)
        loss = torch.tensor(0.0, device=model.device)
        for name, p in model.named_parameters():
            if name in self.fisher:
                loss += (self.fisher[name] * (p - self.params[name]) ** 2).sum()
        return 0.5 * self.lam * loss

def normalize_reward(r):
    global reward_running_mean, reward_running_var, reward_count
    reward_count += 1
    delta = r - reward_running_mean
    reward_running_mean += delta / reward_count
    delta2 = r - reward_running_mean
    reward_running_var += delta * delta2
    std = max(math.sqrt(reward_running_var / max(reward_count, 1)), 1e-8)
    return (r - reward_running_mean) / std

    3. RFT: skip too-easy (>{cfg.rft_max_correct_ratio:.0%}) and too-hard (<{cfg.rft_min_correct_ratio:.0%}) questions
    4. On-policy sampling in RL (model generates its own data)
    5. Replay buffer: {cfg.replay_buffer_ratio:.0%} of batch from base distribution
    6. Temperature sampling (T={cfg.rft_temperature}) for diversity
""")

## Key safeguard summary

| Threat | Mechanism | Where in code |
|---|---|---|
| **Reward hacking** | KL penalty, reward clipping `[-5,5]`, length penalty, repetition detection, reward normalization | `compute_reward()`, PPO `init_kl_coef`/`target_kl`, GRPO `beta` |
| **Catastrophic forgetting** | LoRA (frozen base), EWC regularization, reference KL, low LR in later stages | `EWCRegularizer`, `ref_model`, `rft_lr < sft_lr` |
| **Overfitting** | LoRA dropout, weight decay, early stopping, best-checkpoint loading, grad clipping | `SFTConfig`, `EarlyStoppingCallback` |
| **Distribution mismatch** | OOD eval every N steps, auto-alert on OOD drop, RFT difficulty filtering, temperature diversity | `eval_on_ood_every`, `rft_min/max_correct_ratio` |

> **Data format expected** — one JSONL line per sample:
> ```json
> {"question": "What is 2+2?", "thinking": "2+2 = 4 by addition.", "answer": "4"}
> ```
> Place files at the paths set in `Config` (`data_path`, `eval_data_path`, `ood_data_path`).
