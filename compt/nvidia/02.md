# ── data ──────────────────────────────────────────────────────────────────────

def parse_xml_field(text: str, tag: str) -> str:
    m = re.search(rf"<{tag}>(.*?)</{tag}>", text, re.DOTALL)
    return m.group(1).strip() if m else ""

def load_qa_data(path: str) -> list:
    data = []
    with open(path) as f:
        for line in f:
            obj = json.loads(line.strip())
            if "question" in obj:
                data.append({
                    "question": obj["question"],
                    "thinking": obj.get("thinking", ""),
                    "answer":   obj["answer"],
                })
            elif "text" in obj:
                t = obj["text"]
                data.append({
                    "question": parse_xml_field(t, "question"),
                    "thinking": parse_xml_field(t, "thinking"),
                    "answer":   parse_xml_field(t, "answer"),
                })
    return data

def load_data_safe(path: str, fallback=None) -> list:
    """Load data; if file missing, return fallback or empty list."""
    try:
        return load_qa_data(path)
    except FileNotFoundError:
        print(f"⚠️  {path} not found — using fallback")
        return fallback if fallback else []


# ── model loading ─────────────────────────────────────────────────────────────

def load_base_model_and_tokenizer(config=None):
    c = config or cfg
    bnb = None
    if c.load_in_4bit:
        bnb = BitsAndBytesConfig(
            load_in_4bit=True, bnb_4bit_quant_type="nf4",
            bnb_4bit_compute_dtype=getattr(torch, c.dtype),
            bnb_4bit_use_double_quant=True,
        )
    model = AutoModelForCausalLM.from_pretrained(
        c.base_model_name, quantization_config=bnb,
        device_map=c.device_map, torch_dtype=getattr(torch, c.dtype),
        trust_remote_code=True,
    )
    tok = AutoTokenizer.from_pretrained(c.base_model_name, trust_remote_code=True, padding_side="left")
    if tok.pad_token is None:
        tok.pad_token = tok.eos_token
        model.config.pad_token_id = tok.eos_token_id
    return model, tok


def attach_lora(model, checkpoint_path: str = None, config=None):
    """
    If checkpoint_path is a valid dir with adapter → load it.
    Otherwise create fresh LoRA.
    """
    c = config or cfg
    if checkpoint_path and os.path.isdir(checkpoint_path) and \
       os.path.exists(os.path.join(checkpoint_path, "adapter_config.json")):
        print(f"   📂 Loading existing LoRA from {checkpoint_path}")
        model = PeftModel.from_pretrained(model, checkpoint_path, is_trainable=True)
    else:
        if checkpoint_path:
            print(f"   ⚠️  Checkpoint '{checkpoint_path}' not found — creating fresh LoRA")
        else:
            print("   🆕 Creating fresh LoRA adapter")
        if c.load_in_4bit:
            model = prepare_model_for_kbit_training(model)
        lora_cfg = LoraConfig(
            r=c.lora_r, lora_alpha=c.lora_alpha, lora_dropout=c.lora_dropout,
            target_modules=c.lora_target_modules, bias="none", task_type=TaskType.CAUSAL_LM,
        )
        model = get_peft_model(model, lora_cfg)
    model.print_trainable_parameters()
    return model


def load_model_with_optional_lora(checkpoint_path: str = None, config=None):
    """One-call: base model + optional LoRA → ready to use."""
    c = config or cfg
    model, tok = load_base_model_and_tokenizer(c)
    if checkpoint_path:
        model = attach_lora(model, checkpoint_path, c)
    return model, tok


# ── chat formatting ──────────────────────────────────────────────────────────

def format_chat(question, system_prompt, thinking="", answer=""):
    msgs = [{"role": "system", "content": system_prompt},
            {"role": "user",   "content": question}]
    if answer:
        a = ""
        if thinking:
            a += f"<thinking>\n{thinking}\n</thinking>\n\n"
        a += f"<answer>\n{answer}\n</answer>"
        msgs.append({"role": "assistant", "content": a})
    return msgs

def build_sft_dataset(data, tokenizer, system_prompt):
    rows = []
    for d in data:
        msgs = format_chat(d["question"], system_prompt,
                           thinking=d.get("thinking",""), answer=d["answer"])
        rows.append({"text": tokenizer.apply_chat_template(msgs, tokenize=False)})
    return HFDataset.from_list(rows)


# ── generation ────────────────────────────────────────────────────────────────

@torch.inference_mode()
def generate_answer(model, tokenizer, question, system_prompt,
                    temperature=0.0, max_new_tokens=512):
    msgs = format_chat(question, system_prompt)
    text = tokenizer.apply_chat_template(msgs, tokenize=False, add_generation_prompt=True)
    inp  = tokenizer(text, return_tensors="pt").to(model.device)
    out  = model.generate(**inp, max_new_tokens=max_new_tokens,
                          do_sample=(temperature>0), temperature=max(temperature,1e-7),
                          top_p=0.95, pad_token_id=tokenizer.pad_token_id)
    return tokenizer.decode(out[0, inp["input_ids"].shape[1]:], skip_special_tokens=True)

@torch.inference_mode()
def generate_batch(model, tokenizer, questions, system_prompt,
                   temperature=0.0, max_new_tokens=512, batch_size=8):
    results = []
    for i in range(0, len(questions), batch_size):
        batch = questions[i:i+batch_size]
        texts = [tokenizer.apply_chat_template(
                     format_chat(q, system_prompt), tokenize=False,
                     add_generation_prompt=True) for q in batch]
        inp = tokenizer(texts, return_tensors="pt", padding=True,
                        truncation=True, max_length=cfg.max_seq_length).to(model.device)
        out = model.generate(**inp, max_new_tokens=max_new_tokens,
                             do_sample=(temperature>0), temperature=max(temperature,1e-7),
                             top_p=0.95, pad_token_id=tokenizer.pad_token_id)
        for j, seq in enumerate(out):
            results.append(tokenizer.decode(seq[inp["input_ids"].shape[1]:], skip_special_tokens=True))
    return results


# ── answer checking ───────────────────────────────────────────────────────────

def extract_answer(text):
    ans = parse_xml_field(text, "answer")
    return ans.strip() if ans else text.strip().split("\n")[-1].strip()

def normalize_answer(a):
    a = a.lower().strip()
    a = re.sub(r"\$|\\boxed\{|\}|\\text\{", "", a)
    a = re.sub(r"[^\w\s\.\-]", "", a)
    return a.strip()

def is_correct(pred, gold):
    return normalize_answer(pred) == normalize_answer(gold)


# ── accuracy measurement ─────────────────────────────────────────────────────

def measure_accuracy(model, tokenizer, dataset, system_prompt,
                     temperature=0.0, label="eval", max_samples=None):
    if max_samples:
        dataset = dataset[:max_samples]
    questions = [d["question"] for d in dataset]
    golds     = [d["answer"]   for d in dataset]

    print(f"   Generating {len(questions)} answers for [{label}]…")
    preds = generate_batch(model, tokenizer, questions, system_prompt, temperature=temperature)

    extracted = [extract_answer(p) for p in preds]
    correct   = [is_correct(e, g) for e, g in zip(extracted, golds)]
    fmt_ok    = [("<thinking>" in p and "<answer>" in p) for p in preds]

    acc = sum(correct)/len(correct)
    fmt = sum(fmt_ok)/len(fmt_ok)
    avg_len = np.mean([len(p.split()) for p in preds])

    result = {"label": label, "accuracy": round(acc,4),
              "format_compliance": round(fmt,4),
              "avg_response_words": round(float(avg_len),1),
              "n_samples": len(dataset), "n_correct": sum(correct),
              "timestamp": datetime.now().isoformat()}

    print(f"   ✅  [{label}]  acc={acc:.4f}  fmt={fmt:.4f}  len={avg_len:.0f}")
    return result


# ── reward function ───────────────────────────────────────────────────────────

def compute_reward(response, gold_answer):
    reward = 0.0
    if is_correct(extract_answer(response), gold_answer):
        reward += cfg.reward_correct
    else:
        reward += cfg.reward_incorrect

    if "<thinking>" in response and "</thinking>" in response and \
       "<answer>" in response and "</answer>" in response:
        reward += cfg.reward_format_bonus

    reward -= cfg.reward_length_penalty * len(response.split())

    sents = response.split(".")
    if len(sents) > 3 and len(set(sents))/len(sents) < 0.5:
        reward -= cfg.reward_repetition_penalty

    return reward

def save_step_result(step_name: str, metrics: dict):
    """Save metrics JSON to results_dir.  Filename = step_name sanitised."""

    with open(path, "w") as f:
        json.dump({"step": step_name, **metrics}, f, indent=2)
    print(f"   💾 Saved → {path}")

def load_all_results() -> dict:
    """Read every JSON in results_dir → dict[step_name] = metrics."""
    out = {}
    for fname in sorted(Path(cfg.results_dir).glob("*.json")):
        with open(fname) as f:
            d = json.load(f)
        out[d.get("step", fname.stem)] = d
    return out

# ── EWC helper ────────────────────────────────────────────────────────────────

class EWCRegularizer:
    def __init__(self, model, tokenizer, data, lam=5000.0, n_samples=200):
        self.lam = lam
        self.params, self.fisher = {}, {}
        if not cfg.ewc_enabled:
            return

        print("   Computing Fisher Information for EWC…")
        ds = build_sft_dataset(data[:n_samples], tokenizer, cfg.improved_system_prompt)
        def collate(examples):
            return tokenizer([e["text"] for e in examples], return_tensors="pt",
                             padding=True, truncation=True, max_length=cfg.max_seq_length)
        loader = TorchDL(ds, batch_size=4, collate_fn=collate)

        model.eval()
        fisher_acc = {}
        for n, p in model.named_parameters():
            if p.requires_grad:
                self.params[n] = p.data.clone()
                fisher_acc[n]  = torch.zeros_like(p.data)
        count = 0
        for batch in loader:
            if count >= n_samples: break
            inp = {k: v.to(model.device) for k,v in batch.items()}
            loss = model(**inp, labels=inp.get("input_ids")).loss
            loss.backward(); 
            for n, p in model.named_parameters():
                if p.requires_grad and p.grad is not None:
                    fisher_acc[n] += p.grad.data**2
            model.zero_grad()
            count += inp["input_ids"].shape[0]
        for n in fisher_acc:
            self.fisher[n] = fisher_acc[n] / max(count, 1)
        model.train()

    def penalty(self, model):
        if not self.fisher:
            return torch.tensor(0.0, device=model.device)
        loss = torch.tensor(0.0, device=model.device)
        for n, p in model.named_parameters():
            if n in self.fisher:
                loss += (self.fisher[n] * (p - self.params[n])**2).sum()
        return 0.5 * self.lam * loss

    ppo_prompts.append(tok.apply_chat_template(msgs, tokenize=False, add_generation_prompt=True))
    ppo_golds.append(d["answer"])

# ── Reward normalizer (anti reward-hacking) ──────────────────────────────────

class RunningRewardNorm:
    def __init__(self):
        self.mean, self.var, self.count = 0.0, 1.0, 0
    def __call__(self, r):
        self.count += 1
        d1 = r - self.mean
        self.mean += d1 / self.count
        d2 = r - self.mean
        self.var += d1 * d2
        std = max(math.sqrt(self.var / max(self.count, 1)), 1e-8)
        return max(min((r - self.mean) / std, 5.0), -5.0)

reward_norm = RunningRewardNorm()

# ── Training loop ─────────────────────────────────────────────────────────────

print("🚀  PPO training…")
ppo_reward_log = []

for step in range(cfg.ppo_total_steps):
    idxs = random.sample(range(len(ppo_prompts)), cfg.ppo_batch_size)

    query_tensors = [
        tok(ppo_prompts[i], return_tensors="pt", truncation=True,
            max_length=cfg.max_seq_length)["input_ids"].squeeze(0).to(DEVICE)
        for i in idxs
    ]

    response_tensors = ppo_trainer.generate(
        query_tensors, max_new_tokens=cfg.ppo_max_new_tokens,
        do_sample=True, temperature=0.7, top_p=0.95,
        pad_token_id=tok.pad_token_id,
    )

    responses = [tok.decode(r, skip_special_tokens=True) for r in response_tensors]

    rewards = []
    for resp, i in zip(responses, idxs):
        raw = compute_reward(resp, ppo_golds[i])
        raw -= ewc.penalty(policy_model).item() * 1e-6      # EWC as reward penalty
        rewards.append(torch.tensor(reward_norm(raw), device=DEVICE))

    stats = ppo_trainer.step(query_tensors, response_tensors, rewards)
    avg_r = np.mean([r.item() for r in rewards])
    ppo_reward_log.append(avg_r)

    if (step+1) % 20 == 0:
        print(f"   step {step+1}/{cfg.ppo_total_steps}  "
              f"avg_r={avg_r:.3f}  kl={stats.get('objective/kl',0):.4f}")

    # OOD guard
    if (step+1) % cfg.eval_on_ood_every == 0:
        ood_chk = measure_accuracy(policy_model, tok, ood_data[:20],
                                    cfg.improved_system_prompt,
                                    label=f"PPO-OOD-s{step+1}", max_samples=20)
        # compare with baseline saved on disk (if exists)

        print("⚠️  No results to auto-detect from.  Set EXPORT_CHECKPOINT manually.")

if EXPORT_CHECKPOINT and os.path.isdir(EXPORT_CHECKPOINT):
    os.makedirs(FINAL_DIR, exist_ok=True)
    for f in os.listdir(EXPORT_CHECKPOINT):
        src = os.path.join(EXPORT_CHECKPOINT, f)
        if os.path.isfile(src):
            shutil.copy2(src, os.path.join(FINAL_DIR, f))

    # Save metadata
    meta = {
        "base_model": cfg.base_model_name,
        "source_checkpoint": EXPORT_CHECKPOINT,
        "lora_r": cfg.lora_r, "lora_alpha": cfg.lora_alpha,
        "system_prompt": cfg.improved_system_prompt,
        "all_results": load_all_results(),
    }
    with open(os.path.join(FINAL_DIR, "training_metadata.json"), "w") as f:
        json.dump(meta, f, indent=2, default=str)

    print(f"\n✅  Exported to: {FINAL_DIR}")
    print(f"    Load with:")
    print(f'    >>> model = PeftModel.from_pretrained(base_model, "{FINAL_DIR}")')
else:
    print("❌  No valid checkpoint to export.")

## How the independence works

```
                  ┌────────────────────────────────┐
                  │         CELL 1:  Config         │
                  │  step2_sft_input_checkpoint     │─── None │ any LoRA path
                  │  step3_rft_input_checkpoint     │─── None │ any LoRA path
                  │  step4a_ppo_input_checkpoint    │─── None │ any LoRA path
                  │  step4b_grpo_input_checkpoint   │─── None │ any LoRA path
                  └──────────┬─────────────────────┘
                             │  each step reads ONLY its own knob
          ┌──────────────────┼──────────────────┬──────────────────┐
          ▼                  ▼                  ▼                  ▼
     ┌─────────┐       ┌─────────┐       ┌──────────┐      ┌──────────┐
     │ STEP 1  │       │ STEP 2  │       │ STEP 4a  │      │ STEP 4b  │
     │ Prompt  │       │  SFT    │       │   PPO    │      │   GRPO   │
     │(no LoRA)│       │         │       │          │      │          │
     └────┬────┘       └────┬────┘       └────┬─────┘      └────┬─────┘
          │                 │                  │                 │
          ▼                 ▼                  ▼                 ▼
    results/*.json    results/*.json     results/*.json    results/*.json
                  ↘        ↓        ↙            ↓
                   ┌───────────────────┐
                   │   COMPARE  cell   │   ← reads all JSONs
                   │  (run any time)   │
                   └───────────────────┘
```

**Example workflows:**

| Want to do | Set in Config | Run cells |
|---|---|---|
| Only SFT | `step2_input = None` | Cell 1→2→3→**Step 2**→Compare |
| Only GRPO from base | `step4b_input = None` | Cell 1→2→3→**Step 4b**→Compare |
| SFT then PPO | `step4a_input = "outputs/step2_sft"` | **Step 2** first, then **Step 4a** |
| Compare PPO vs GRPO both from SFT | `step4a_input = step4b_input = "outputs/step2_sft"` | **Step 4a** + **Step 4b** in any order → **Compare** |
| RFT from an external adapter | `step3_input = "/my/other/adapter"` | **Step 3** alone |
| Re-run Step 1 after training | `STEP1_CHECKPOINT = "outputs/step4b_grpo"` | **Step 1** alone |
